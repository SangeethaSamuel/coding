{
  int lmax=arr[0];
  int rmax=arr[n-1];
  int left=lmax+1;
  int right=rmax+1;
  int res = 0;
  while(left<right)
  {
    if(lmax<rmax)
    {
      res+=max(0,lmax-arr[left]);
      lmax=max(lmax,arr[left]);
      left+=1 ;
    }
    else
    {
      res+=max(0,rmax-arr[right])
      rmax=max(rmax,arr[right]);
      right-=1;
    }
  }
  return res;
}
